cmake_minimum_required(VERSION 2.8.12)
project(delphi)

set(CMAKE_CXX_STANDARD 17)
set(Boost_INCLUDE_DIR /usr/local/Cellar/boost/1.70.0/include)
set(Boost_LIBRARY_DIR /usr/local/Cellar/boost/1.70.0/lib)
find_package(Boost)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_SOURCE_DIR}/delphi/cpp)
set(PYBIND11_CPP_STANDARD -std=c++17)
set(PYBIND11_PYTHON_VERSION 3.7)
add_subdirectory(external/pybind11)
include_directories(lib)
include_directories(external)
pybind11_add_module(kde lib/kde_wrapper.cpp)
pybind11_add_module(AnalysisGraph lib/AnalysisGraph_wrapper.cpp)

# Finding and linking SQLite3
include(FindPackageHandleStandardArgs)
set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${CMAKE_CURRENT_SOURCE_DIR}/cmake")
find_package(SQLite3)
if (SQLITE3_FOUND)
  include_directories(${SQLITE3_INCLUDE_DIRS})
  target_link_libraries(AnalysisGraph PUBLIC ${SQLITE3_LIBRARIES})
endif (SQLITE3_FOUND)
if(Boost_FOUND)
  include_directories(${Boost_INCLUDE_DIRS})
  link_directories(${Boost_LIBRARY_DIR})
  target_link_libraries(kde ${Boost_LIBRARIES})
  target_link_libraries(AnalysisGraph ${Boost_LIBRARIES})
endif(Boost_FOUND)
